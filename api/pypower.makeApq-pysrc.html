 29 -def makeApq(baseMVA, gen): 

 30      """Construct linear constraints for generator capability curves. 
 31   
 32      Constructs the parameters for the following linear constraints 
 33      implementing trapezoidal generator capability curves, where 
 34      C{Pg} and C{Qg} are the real and reactive generator injections:: 
 35   
 36          Apqh * [Pg, Qg] <= ubpqh 
 37          Apql * [Pg, Qg] <= ubpql 
 38   
 39      C{data} constains additional information as shown below. 
 40   
 41      Example:: 
 42          Apqh, ubpqh, Apql, ubpql, data = makeApq(baseMVA, gen) 
 43   
 44          data['h']      [Qc1max-Qc2max, Pc2-Pc1] 
 45          data['l']      [Qc2min-Qc1min, Pc1-Pc2] 
 46          data['ipqh']   indices of gens with general PQ cap curves (upper) 
 47          data['ipql']   indices of gens with general PQ cap curves (lower) 
 48      """ 
 49      data = {} 
 50      ## data dimensions 
 51      ng = gen.shape[0]      ## number of dispatchable injections 
 52   
 53      ## which generators require additional linear constraints 
 54      ## (in addition to simple box constraints) on (Pg,Qg) to correctly 
 55      ## model their PQ capability curves 
 56      ipqh = find( hasPQcap(gen, 'U') ) 
 57      ipql = find( hasPQcap(gen, 'L') ) 
 58      npqh = ipqh.shape[0]   ## number of general PQ capability curves (upper) 
 59      npql = ipql.shape[0]   ## number of general PQ capability curves (lower) 
 60   
 61      ## make Apqh if there is a need to add general PQ capability curves 
 62      ## use normalized coefficient rows so multipliers have right scaling 
 63      ## in $$/pu 
 64      if npqh > 0: 
 65          data["h"] = c_[gen[ipqh, QC1MAX] - gen[ipqh, QC2MAX], 
 66                         gen[ipqh, PC2] - gen[ipqh, PC1]] 
 67          ubpqh = data["h"][:, 0] * gen[ipqh, PC1] + \ 
 68                  data["h"][:, 1] * gen[ipqh, QC1MAX] 
 69          for i in range(npqh): 
 70              tmp = linalg.norm(data["h"][i, :]) 
 71              data["h"][i, :] = data["h"][i, :] / tmp 
 72              ubpqh[i] = ubpqh[i] / tmp 
 73          Apqh = sparse((data["h"].flatten('F'), 
 74                         (r_[arange(npqh), arange(npqh)], r_[ipqh, ipqh+ng])), 
 75                        (npqh, 2*ng)) 
 76          ubpqh = ubpqh / baseMVA 
 77      else: 
 78          data["h"] = array([]) 
 79          Apqh  = zeros((0, 2*ng)) 
 80          ubpqh = array([]) 
 81   
 82      ## similarly Apql 
 83      if npql > 0: 
 84          data["l"] = c_[gen[ipql, QC2MIN] - gen[ipql, QC1MIN], 
 85                         gen[ipql, PC1] - gen[ipql, PC2]] 
 86          ubpql = data["l"][:, 0] * gen[ipql, PC1] + \ 
 87                  data["l"][:, 1] * gen[ipql, QC1MIN] 
 88          for i in range(npql): 
 89              tmp = linalg.norm(data["l"][i, :]) 
 90              data["l"][i, :] = data["l"][i, :] / tmp 
 91              ubpql[i] = ubpql[i] / tmp 
 92          Apql = sparse((data["l"].flatten('F'), 
 93                         (r_[arange(npql), arange(npql)], r_[ipql, ipql+ng])), 
 94                        (npql, 2*ng)) 
 95          ubpql = ubpql / baseMVA 
 96      else: 
 97          data["l"] = array([]) 
 98          Apql  = zeros((0, 2*ng)) 
 99          ubpql = array([]) 
100   
101      data["ipql"] = ipql 
102      data["ipqh"] = ipqh 
103   
104      return Apqh, ubpqh, Apql, ubpql, data
