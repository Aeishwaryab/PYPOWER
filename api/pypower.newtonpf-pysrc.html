 32      """Solves the power flow using a full Newton's method. 
 33   
 34      Solves for bus voltages given the full system admittance matrix (for 
 35      all buses), the complex bus power injection vector (for all buses), 
 36      the initial vector of complex bus voltages, and column vectors with 
 37      the lists of bus indices for the swing bus, PV buses, and PQ buses, 
 38      respectively. The bus voltage vector contains the set point for 
 39      generator (including ref bus) buses, and the reference angle of the 
 40      swing bus, as well as an initial guess for remaining magnitudes and 
 41      angles. C{ppopt} is a PYPOWER options vector which can be used to 
 42      set the termination tolerance, maximum number of iterations, and 
 43      output options (see L{ppoption} for details). Uses default options if 
 44      this parameter is not given. Returns the final complex voltages, a 
 45      flag which indicates whether it converged or not, and the number of 
 46      iterations performed. 
 47   
 48      @see: L{runpf} 
 49      """ 
 50      ## default arguments 
 51      if ppopt is None: 
 52          ppopt = ppoption() 
 53   
 54      ## options 
 55      tol     = ppopt['PF_TOL'] 
 56      max_it  = ppopt['PF_MAX_IT'] 
 57      verbose = ppopt['VERBOSE'] 
 58   
 59      ## initialize 
 60      converged = 0 
 61      i = 0 
 62      V = V0 
 63      Va = angle(V) 
 64      Vm = abs(V) 
 65   
 66      ## set up indexing for updating V 
 67      pvpq = r_[pv, pq] 
 68      npv = len(pv) 
 69      npq = len(pq) 
 70      j1 = 0;         j2 = npv           ## j1:j2 - V angle of pv buses 
 71      j3 = j2;        j4 = j2 + npq      ## j3:j4 - V angle of pq buses 
 72      j5 = j4;        j6 = j4 + npq      ## j5:j6 - V mag of pq buses 
 73   
 74      ## evaluate F(x0) 
 75      mis = V * conj(Ybus * V) - Sbus 
 76      F = r_[  mis[pv].real, 
 77               mis[pq].real, 
 78               mis[pq].imag  ] 
 79   
 80      ## check tolerance 
 81      normF = linalg.norm(F, Inf) 
 82      if verbose > 0: 
 83          sys.stdout.write('(Newton)\n') 
 84      if verbose > 1: 
 85          sys.stdout.write('\n it    max P & Q mismatch (p.u.)') 
 86          sys.stdout.write('\n----  ---------------------------') 
 87          sys.stdout.write('\n%3d        %10.3e' % (i, normF)) 
 88      if normF < tol: 
 89          converged = 1 
 90          if verbose > 1: 
 91              sys.stdout.write('\nConverged!\n') 
 92   
 93      ## do Newton iterations 
 94      while (not converged and i < max_it): 
 95          ## update iteration counter 
 96          i = i + 1 
 97   
 98          ## evaluate Jacobian 
 99          dS_dVm, dS_dVa = dSbus_dV(Ybus, V) 
100   
101          J11 = dS_dVa[array([pvpq]).T, pvpq].real 
102          J12 = dS_dVm[array([pvpq]).T, pq].real 
103          J21 = dS_dVa[array([pq]).T, pvpq].imag 
104          J22 = dS_dVm[array([pq]).T, pq].imag 
105   
106          J = vstack([ 
107                  hstack([J11, J12]), 
108                  hstack([J21, J22]) 
109              ], format="csr") 
110   
111          ## compute update step 
112          dx = -1 * spsolve(J, F) 
113   
114          ## update voltage 
115          if npv: 
116              Va[pv] = Va[pv] + dx[j1:j2] 
117          if npq: 
118              Va[pq] = Va[pq] + dx[j3:j4] 
119              Vm[pq] = Vm[pq] + dx[j5:j6] 
120          V = Vm * exp(1j * Va) 
121          Vm = abs(V)            ## update Vm and Va again in case 
122          Va = angle(V)          ## we wrapped around with a negative Vm 
123   
124          ## evalute F(x) 
125          mis = V * conj(Ybus * V) - Sbus 
126          F = r_[  mis[pv].real, 
127                   mis[pq].real, 
128                   mis[pq].imag  ] 
129   
130          ## check for convergence 
131          normF = linalg.norm(F, Inf) 
132          if verbose > 1: 
133              sys.stdout.write('\n%3d        %10.3e' % (i, normF)) 
134          if normF < tol: 
135              converged = 1 
136              if verbose: 
137                  sys.stdout.write("\nNewton's method power flow converged in " 
138                                   "%d iterations.\n" % i) 
139   
140      if verbose: 
141          if not converged: 
142              sys.stdout.write("\nNewton's method power did not converge in %d " 
143                               "iterations.\n" % i) 
144   
145      return V, converged, i
