 47      Assumes that ppc is a PYPOWER case dict with internal indexing, 
 48      all equipment in-service, etc. 
 49   
 50      @see: L{opf}, L{ext2int}, L{opf_execute} 
 51      """ 
 52      ## options 
 53      dc  = ppopt['PF_DC']        ## 1 = DC OPF, 0 = AC OPF 
 54      alg = ppopt['OPF_ALG'] 
 55      verbose = ppopt['VERBOSE'] 
 56   
 57      ## data dimensions 
 58      nb = ppc['bus'].shape[0]    ## number of buses 
 59      nl = ppc['branch'].shape[0] ## number of branches 
 60      ng = ppc['gen'].shape[0]    ## number of dispatchable injections 
 61      if 'A' in ppc: 
 62          nusr = ppc['A'].shape[0]    ## number of linear user constraints 
 63      else: 
 64          nusr = 0 
 65   
 66      if 'N' in ppc: 
 67          nw = ppc['N'].shape[0]      ## number of general cost vars, w 
 68      else: 
 69          nw = 0 
 70   
 71      if dc: 
 72          ## ignore reactive costs for DC 
 73          ppc['gencost'], _ = pqcost(ppc['gencost'], ng) 
 74   
 75          ## reduce A and/or N from AC dimensions to DC dimensions, if needed 
 76          if nusr or nw: 
 77              acc = r_[nb + arange(nb), 2 * nb + ng + arange(ng)]   ## Vm and Qg columns 
 78   
 79              if nusr and (ppc['A'].shape[1] >= 2*nb + 2*ng): 
 80                  ## make sure there aren't any constraints on Vm or Qg 
 81                  if ppc['A'][:, acc].nnz > 0: 
 82                      stderr.write('opf_setup: attempting to solve DC OPF with user constraints on Vm or Qg\n') 
 83   
 84                  # FIXME: delete sparse matrix columns 
 85                  bcc = delete(arange(ppc['A'].shape[1]), acc) 
 86                  ppc['A'] = ppc['A'].tolil()[:, bcc].tocsr()           ## delete Vm and Qg columns 
 87   
 88              if nw and (ppc['N'].shape[1] >= 2*nb + 2*ng): 
 89                  ## make sure there aren't any costs on Vm or Qg 
 90                  if ppc['N'][:, acc].nnz > 0: 
 91                      ii, _ = nonzero(ppc['N'][:, acc]) 
 92                      _, ii = unique(ii, return_index=True)    ## indices of w with potential non-zero cost terms from Vm or Qg 
 93                      if any(ppc['Cw'][ii]) | ( ('H' in ppc) & (len(ppc['H']) > 0) & 
 94                              any(any(ppc['H'][:, ii])) ): 
 95                          stderr.write('opf_setup: attempting to solve DC OPF with user costs on Vm or Qg\n') 
 96   
 97                  # FIXME: delete sparse matrix columns 
 98                  bcc = delete(arange(ppc['N'].shape[1]), acc) 
 99                  ppc['N'] = ppc['N'].tolil()[:, bcc].tocsr()               ## delete Vm and Qg columns

101      ## convert single-block piecewise-linear costs into linear polynomial cost 
102      pwl1 = find((ppc['gencost'][:, MODEL] == PW_LINEAR) & (ppc['gencost'][:, NCOST] == 2)) 
103      # p1 = array([]) 
104      if len(pwl1) > 0: 
105          x0 = ppc['gencost'][pwl1, COST] 
106          y0 = ppc['gencost'][pwl1, COST + 1] 
107          x1 = ppc['gencost'][pwl1, COST + 2] 
108          y1 = ppc['gencost'][pwl1, COST + 3] 
109          m = (y1 - y0) / (x1 - x0) 
110          b = y0 - m * x0 
111          ppc['gencost'][pwl1, MODEL] = POLYNOMIAL 
112          ppc['gencost'][pwl1, NCOST] = 2 
113          ppc['gencost'][pwl1, COST:COST + 2] = r_[m, b]

116      baseMVA, bus, gen, branch, gencost, _, lbu, ubu, ppopt, \ 
117              _, fparm, H, Cw, z0, zl, zu, userfcn, _ = opf_args(ppc, ppopt)

119      ## warn if there is more than one reference bus 
120      refs = find(bus[:, BUS_TYPE] == REF) 
121      if len(refs) > 1 and verbose > 0: 
122          errstr = '\nopf_setup: Warning: Multiple reference buses.\n' + \ 
123              '           For a system with islands, a reference bus in each island\n' + \ 
124              '           may help convergence, but in a fully connected system such\n' + \ 
125              '           a situation is probably not reasonable.\n\n'

131      Vm   = bus[:, VM].copy() 
132      Vm[gbus] = gen[:, VG]   ## buses with gens, init Vm from gen data 
133      Pg   = gen[:, PG] / baseMVA 
134      Qg   = gen[:, QG] / baseMVA 
135      Pmin = gen[:, PMIN] / baseMVA 
136      Pmax = gen[:, PMAX] / baseMVA 
137      Qmin = gen[:, QMIN] / baseMVA 
138      Qmax = gen[:, QMAX] / baseMVA

140      if dc:               ## DC model 
141          ## more problem dimensions 
142          nv    = 0            ## number of voltage magnitude vars 
143          nq    = 0            ## number of Qg vars 
144          q1    = array([])    ## index of 1st Qg column in Ay 
145   
146          ## power mismatch constraints 
147          B, Bf, Pbusinj, Pfinj = makeBdc(baseMVA, bus, branch) 
148          neg_Cg = sparse((-ones(ng), (gen[:, GEN_BUS], arange(ng))), (nb, ng))   ## Pbus w.r.t. Pg

157          upt = branch[il, RATE_A] / baseMVA + Pfinj[il]

169   
170          ## generator PQ capability curve constraints 
171          Apqh, ubpqh, Apql, ubpql, Apqdata = makeApq(baseMVA, gen)

173          user_vars = ['Va', 'Vm', 'Pg', 'Qg'] 
174          ycon_vars = ['Pg', 'Qg', 'y']

176      ## voltage angle reference constraints 
177      Vau = Inf * ones(nb) 
178      Val = -Vau 
179      Vau[refs] = Va[refs]

182      ## branch voltage angle difference limits 
183      Aang, lang, uang, iang  = makeAang(baseMVA, branch, nb, ppopt)

185      ## basin constraints for piece-wise linear gen cost variables

195      if any((gencost[:, MODEL] != POLYNOMIAL) & (gencost[:, MODEL] != PW_LINEAR)):

203              stderr.write('opf_setup: user supplied A matrix must have at least %d columns.\n' % nx) 
204      else: 
205          nz = 0               ## number of user z variables 
206          if nw:               ## still need to check number of columns of N 
207              if ppc['N'].shape[1] != nx: 
208                  stderr.write('opf_setup: user supplied N matrix must have %d columns.\n' % nx)

211      om = opf_model(ppc) 
212      if len(pwl1) > 0: 
213          om.userdata('pwl1', pwl1) 
214   
215      if dc: 
216          om.userdata('Bf', Bf) 
217          om.userdata('Pfinj', Pfinj) 
218          om.userdata('iang', iang) 
219          om.add_vars('Va', nb, Va, Val, Vau) 
220          om.add_vars('Pg', ng, Pg, Pmin, Pmax) 
221          om.add_constraints('Pmis', Amis, bmis, bmis, ['Va', 'Pg']) ## nb 
222          om.add_constraints('Pf',  Bf[il, :], lpf, upf, ['Va'])     ## nl 
223          om.add_constraints('Pt', -Bf[il, :], lpf, upt, ['Va'])     ## nl 
224          om.add_constraints('ang', Aang, lang, uang, ['Va'])        ## nang 
225      else: 
226          om.userdata('Apqdata', Apqdata) 
227          om.userdata('iang', iang) 
228          om.add_vars('Va', nb, Va, Val, Vau) 
229          om.add_vars('Vm', nb, Vm, bus[:, VMIN], bus[:, VMAX]) 
230          om.add_vars('Pg', ng, Pg, Pmin, Pmax) 
231          om.add_vars('Qg', ng, Qg, Qmin, Qmax) 
232          om.add_constraints('Pmis', nb, 'nonlinear') 
233          om.add_constraints('Qmis', nb, 'nonlinear') 
234          om.add_constraints('Sf', nl, 'nonlinear') 
235          om.add_constraints('St', nl, 'nonlinear') 
236          om.add_constraints('PQh', Apqh, array([]), ubpqh, ['Pg', 'Qg'])   ## npqh 
237          om.add_constraints('PQl', Apql, array([]), ubpql, ['Pg', 'Qg'])   ## npql 
238          om.add_constraints('vl',  Avl, lvl, uvl,   ['Pg', 'Qg'])   ## nvl 
239          om.add_constraints('ang', Aang, lang, uang, ['Va'])        ## nang 
240   
241      ## y vars, constraints for piece-wise linear gen costs 
242      if ny > 0: 
243          om.add_vars('y', ny) 
244          om.add_constraints('ycon', Ay, array([]), by, ycon_vars)          ## ncony 
245   
246      ## add user vars, constraints and costs (as specified via A, ..., N, ...) 
247      if nz > 0: 
248          om.add_vars('z', nz, z0, zl, zu) 
249          user_vars.append('z') 
250   
251      if nusr: 
252          om.add_constraints('usr', ppc['A'], lbu, ubu, user_vars)      ## nusr 
253   
254      if nw: 
255          user_cost = {} 
256          user_cost['N'] = ppc['N'] 
257          user_cost['Cw'] = Cw 
258          if len(fparm) > 0:

261              user_cost['kk'] = fparm[:, 2]

267          om.add_costs('usr', user_cost, user_vars)
