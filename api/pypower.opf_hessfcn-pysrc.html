 38      """Evaluates Hessian of Lagrangian for AC OPF. 
 39   
 40      Hessian evaluation function for AC optimal power flow, suitable 
 41      for use with L{pips}. 
 42   
 43      Examples:: 
 44          Lxx = opf_hessfcn(x, lmbda, om, Ybus, Yf, Yt, ppopt) 
 45          Lxx = opf_hessfcn(x, lmbda, om, Ybus, Yf, Yt, ppopt, il) 
 46          Lxx = opf_hessfcn(x, lmbda, om, Ybus, Yf, Yt, ppopt, il, cost_mult) 
 47   
 48      @param x: optimization vector 
 49      @param lmbda: C{eqnonlin} - Lagrange multipliers on power balance 
 50      equations. C{ineqnonlin} - Kuhn-Tucker multipliers on constrained 
 51      branch flows. 
 52      @param om: OPF model object 
 53      @param Ybus: bus admittance matrix 
 54      @param Yf: admittance matrix for "from" end of constrained branches 
 55      @param Yt: admittance matrix for "to" end of constrained branches 
 56      @param ppopt: PYPOWER options vector 
 57      @param il: (optional) vector of branch indices corresponding to 
 58      branches with flow limits (all others are assumed to be unconstrained). 
 59      The default is C{range(nl)} (all branches). C{Yf} and C{Yt} contain 
 60      only the rows corresponding to C{il}. 
 61      @param cost_mult: (optional) Scale factor to be applied to the cost 
 62      (default = 1). 
 63   
 64      @return: Hessian of the Lagrangian. 
 65   
 66      @see: L{opf_costfcn}, L{opf_consfcn} 
 67      """ 
 68      ##----- initialize ----- 
 69      ## unpack data 
 70      ppc = om.get_ppc() 
 71      baseMVA, bus, gen, branch, gencost = \ 
 72          ppc["baseMVA"], ppc["bus"], ppc["gen"], ppc["branch"], ppc["gencost"] 
 73      cp = om.get_cost_params() 
 74      N, Cw, H, dd, rh, kk, mm = \ 
 75          cp["N"], cp["Cw"], cp["H"], cp["dd"], cp["rh"], cp["kk"], cp["mm"] 
 76      vv, _, _, _ = om.get_idx() 
 77   
 78      ## unpack needed parameters 
 79      nb = bus.shape[0]          ## number of buses 
 80      nl = branch.shape[0]       ## number of branches 
 81      ng = gen.shape[0]          ## number of dispatchable injections 
 82      nxyz = len(x)              ## total number of control vars of all types 
 83   
 84      ## set default constrained lines 
 85      if il is None: 
 86          il = arange(nl)            ## all lines have limits by default 
 87      nl2 = len(il)           ## number of constrained lines 
 88   
 89      ## grab Pg & Qg 
 90      Pg = x[vv["i1"]["Pg"]:vv["iN"]["Pg"]]  ## active generation in p.u. 
 91      Qg = x[vv["i1"]["Qg"]:vv["iN"]["Qg"]]  ## reactive generation in p.u. 
 92   
 93      ## put Pg & Qg back in gen 
 94      gen[:, PG] = Pg * baseMVA  ## active generation in MW 
 95      gen[:, QG] = Qg * baseMVA  ## reactive generation in MVAr 
 96   
 97      ## reconstruct V 
 98      Va = x[vv["i1"]["Va"]:vv["iN"]["Va"]] 
 99      Vm = x[vv["i1"]["Vm"]:vv["iN"]["Vm"]] 
100      V = Vm * exp(1j * Va) 
101      nxtra = nxyz - 2 * nb 
102      pcost = gencost[arange(ng), :] 
103      if gencost.shape[0] > ng: 
104          qcost = gencost[arange(ng, 2 * ng), :] 
105      else: 
106          qcost = array([]) 
107   
108      ## ----- evaluate d2f ----- 
109      d2f_dPg2 = zeros(ng)#sparse((ng, 1))               ## w.r.t. p.u. Pg 
110      d2f_dQg2 = zeros(ng)#sparse((ng, 1))               ## w.r.t. p.u. Qg 
111      ipolp = find(pcost[:, MODEL] == POLYNOMIAL) 
112      d2f_dPg2[ipolp] = \ 
113              baseMVA**2 * polycost(pcost[ipolp, :], Pg[ipolp] * baseMVA, 2) 
114      if any(qcost):          ## Qg is not free 
115          ipolq = find(qcost[:, MODEL] == POLYNOMIAL) 
116          d2f_dQg2[ipolq] = \ 
117                  baseMVA**2 * polycost(qcost[ipolq, :], Qg[ipolq] * baseMVA, 2) 
118      i = r_[arange(vv["i1"]["Pg"], vv["iN"]["Pg"]), 
119             arange(vv["i1"]["Qg"], vv["iN"]["Qg"])] 
120  #    d2f = sparse((vstack([d2f_dPg2, d2f_dQg2]).toarray().flatten(), 
121  #                  (i, i)), shape=(nxyz, nxyz)) 
122      d2f = sparse((r_[d2f_dPg2, d2f_dQg2], (i, i)), (nxyz, nxyz)) 
123   
124      ## generalized cost 
125      if issparse(N) and N.nnz > 0: 
126          nw = N.shape[0] 
127          r = N * x - rh                    ## Nx - rhat 
128          iLT = find(r < -kk)               ## below dead zone 
129          iEQ = find((r == 0) & (kk == 0))  ## dead zone doesn't exist 
130          iGT = find(r > kk)                ## above dead zone 
131          iND = r_[iLT, iEQ, iGT]           ## rows that are Not in the Dead region 
132          iL = find(dd == 1)                ## rows using linear function 
133          iQ = find(dd == 2)                ## rows using quadratic function 
134          LL = sparse((ones(len(iL)), (iL, iL)), (nw, nw)) 
135          QQ = sparse((ones(len(iQ)), (iQ, iQ)), (nw, nw)) 
136          kbar = sparse((r_[ones(len(iLT)), zeros(len(iEQ)), -ones(len(iGT))], 
137                         (iND, iND)), (nw, nw)) * kk 
138          rr = r + kbar                  ## apply non-dead zone shift 
139          M = sparse((mm[iND], (iND, iND)), (nw, nw))  ## dead zone or scale 
140          diagrr = sparse((rr, (arange(nw), arange(nw))), (nw, nw)) 
141   
142          ## linear rows multiplied by rr(i), quadratic rows by rr(i)^2 
143          w = M * (LL + QQ * diagrr) * rr 
144          HwC = H * w + Cw 
145          AA = N.T * M * (LL + 2 * QQ * diagrr) 
146   
147          d2f = d2f + AA * H * AA.T + 2 * N.T * M * QQ * \ 
148                  sparse((HwC, (arange(nw), arange(nw))), (nw, nw)) * N 
149      d2f = d2f * cost_mult 
150   
151      ##----- evaluate Hessian of power balance constraints ----- 
152      nlam = len(lmbda["eqnonlin"]) / 2 
153      lamP = lmbda["eqnonlin"][:nlam] 
154      lamQ = lmbda["eqnonlin"][nlam:nlam + nlam] 
155      Gpaa, Gpav, Gpva, Gpvv = d2Sbus_dV2(Ybus, V, lamP) 
156      Gqaa, Gqav, Gqva, Gqvv = d2Sbus_dV2(Ybus, V, lamQ) 
157   
158      d2G = vstack([ 
159              hstack([ 
160                  vstack([hstack([Gpaa, Gpav]), 
161                          hstack([Gpva, Gpvv])]).real + 
162                  vstack([hstack([Gqaa, Gqav]), 
163                          hstack([Gqva, Gqvv])]).imag, 
164                  sparse((2 * nb, nxtra))]), 
165              hstack([ 
166                  sparse((nxtra, 2 * nb)), 
167                  sparse((nxtra, nxtra)) 
168              ]) 
169          ], "csr") 
170   
171      ##----- evaluate Hessian of flow constraints ----- 
172      nmu = len(lmbda["ineqnonlin"]) / 2 
173      muF = lmbda["ineqnonlin"][:nmu] 
174      muT = lmbda["ineqnonlin"][nmu:nmu + nmu] 
175      if ppopt['OPF_FLOW_LIM'] == 2:       ## current 
176          dIf_dVa, dIf_dVm, dIt_dVa, dIt_dVm, If, It = dIbr_dV(Yf, Yt, V) 
177          Hfaa, Hfav, Hfva, Hfvv = d2AIbr_dV2(dIf_dVa, dIf_dVm, If, Yf, V, muF) 
178          Htaa, Htav, Htva, Htvv = d2AIbr_dV2(dIt_dVa, dIt_dVm, It, Yt, V, muT) 
179      else: 
180          f = branch[il, F_BUS].astype(int)    ## list of "from" buses 
181          t = branch[il, T_BUS].astype(int)    ## list of "to" buses 
182          ## connection matrix for line & from buses 
183          Cf = sparse((ones(nl2), (arange(nl2), f)), (nl2, nb)) 
184          ## connection matrix for line & to buses 
185          Ct = sparse((ones(nl2), (arange(nl2), t)), (nl2, nb)) 
186          dSf_dVa, dSf_dVm, dSt_dVa, dSt_dVm, Sf, St = \ 
187                  dSbr_dV(branch[il,:], Yf, Yt, V) 
188          if ppopt['OPF_FLOW_LIM'] == 1:     ## real power 
189              Hfaa, Hfav, Hfva, Hfvv = d2ASbr_dV2(dSf_dVa.real, dSf_dVm.real, 
190                                                  Sf.real, Cf, Yf, V, muF) 
191              Htaa, Htav, Htva, Htvv = d2ASbr_dV2(dSt_dVa.real, dSt_dVm.real, 
192                                                  St.real, Ct, Yt, V, muT) 
193          else:                  ## apparent power 
194              Hfaa, Hfav, Hfva, Hfvv = \ 
195                      d2ASbr_dV2(dSf_dVa, dSf_dVm, Sf, Cf, Yf, V, muF) 
196              Htaa, Htav, Htva, Htvv = \ 
197                      d2ASbr_dV2(dSt_dVa, dSt_dVm, St, Ct, Yt, V, muT) 
198   
199      d2H = vstack([ 
200              hstack([ 
201                  vstack([hstack([Hfaa, Hfav]), 
202                          hstack([Hfva, Hfvv])]) + 
203                  vstack([hstack([Htaa, Htav]), 
204                          hstack([Htva, Htvv])]), 
205                  sparse((2 * nb, nxtra)) 
206              ]), 
207              hstack([ 
208                  sparse((nxtra, 2 * nb)), 
209                  sparse((nxtra, nxtra)) 
210              ]) 
211          ], "csr") 
212   
213      ##-----  do numerical check using (central) finite differences  ----- 
214      if 0: 
215          nx = len(x) 
216          step = 1e-5 
217          num_d2f = sparse((nx, nx)) 
218          num_d2G = sparse((nx, nx)) 
219          num_d2H = sparse((nx, nx)) 
220          for i in range(nx): 
221              xp = x 
222              xm = x 
223              xp[i] = x[i] + step / 2 
224              xm[i] = x[i] - step / 2 
225              # evaluate cost & gradients 
226              _, dfp = opf_costfcn(xp, om) 
227              _, dfm = opf_costfcn(xm, om) 
228              # evaluate constraints & gradients 
229              _, _, dHp, dGp = opf_consfcn(xp, om, Ybus, Yf, Yt, ppopt, il) 
230              _, _, dHm, dGm = opf_consfcn(xm, om, Ybus, Yf, Yt, ppopt, il) 
231              num_d2f[:, i] = cost_mult * (dfp - dfm) / step 
232              num_d2G[:, i] = (dGp - dGm) * lmbda["eqnonlin"]   / step 
233              num_d2H[:, i] = (dHp - dHm) * lmbda["ineqnonlin"] / step 
234          d2f_err = max(max(abs(d2f - num_d2f))) 
235          d2G_err = max(max(abs(d2G - num_d2G))) 
236          d2H_err = max(max(abs(d2H - num_d2H))) 
237          if d2f_err > 1e-6: 
238              print 'Max difference in d2f: %g' % d2f_err 
239          if d2G_err > 1e-5: 
240              print 'Max difference in d2G: %g' % d2G_err 
241          if d2H_err > 1e-6: 
242              print 'Max difference in d2H: %g' % d2H_err 
243   
244      return d2f + d2G + d2H
