 37   
 38              min f(x) 
 39               x 
 40

 42

 47   
 48      Note: The calling syntax is almost identical to that of FMINCON from 
 49      MathWorks' Optimization Toolbox. The main difference is that the linear 
 50      constraints are specified with C{A}, C{L}, C{U} instead of C{A}, C{B}, 
 51      C{Aeq}, C{Beq}. The functions for evaluating the objective function, 
 52      constraints and Hessian are identical. 
 53   
 54      Example from U{http://en.wikipedia.org/wiki/Nonlinear_programming}:

 56          >>> from scipy.sparse import csr_matrix

 59          ...     df = -r_[x[1], x[0] + x[2], x[1]] 
 60          ...     # actually not used since 'hess_fcn' is provided 
 61          ...     d2f = -array([[0, 1, 0], [1, 0, 1], [0, 1, 0]], float64) 
 62          ...     return f, df, d2f 
 63          >>> def gh2(x): 
 64          ...     h = dot(array([[1, -1, 1], 
 65          ...                    [1,  1, 1]]), x**2) + array([-2.0, -10.0]) 
 66          ...     dh = 2 * csr_matrix(array([[ x[0], x[0]], 
 67          ...                                [-x[1], x[1]], 
 68          ...                                [ x[2], x[2]]])) 
 69          ...     g = array([]) 
 70          ...     dg = None 
 71          ...     return h, g, dh, dg

 73          ...     mu = lam["ineqnonlin"] 
 74          ...     a = r_[dot(2 * array([1, 1]), mu), -1, 0] 
 75          ...     b = r_[-1, dot(2 * array([-1, 1]), mu),-1] 
 76          ...     c = r_[0, -1, dot(2 * array([1, 1]), mu)] 
 77          ...     Lxx = csr_matrix(array([a, b, c])) 
 78          ...     return Lxx
