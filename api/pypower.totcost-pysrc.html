32      same dimensions as PG. Each row of C{gencost} is used to evaluate the 
33      cost at the points specified in the corresponding row of C{Pg}. 
34      """ 
35      ng, m = gencost.shape 
36      totalcost = zeros(ng) 
37   
38      if len(gencost) > 0: 
39          ipwl = find(gencost[:, MODEL] == PW_LINEAR) 
40          ipol = find(gencost[:, MODEL] == POLYNOMIAL) 
41          if len(ipwl) > 0: 
42              p = gencost[:, COST:(m-1):2] 
43              c = gencost[:, (COST+1):m:2] 
44   
45              for i in ipwl: 
46                  ncost = gencost[i, NCOST] 
47                  for k in arange(ncost - 1): 
48                      p1, p2 = p[i, k], p[i, k+1] 
49                      c1, c2 = c[i, k], c[i, k+1] 
50                      m = (c2 - c1) / (p2 - p1) 
51                      b = c1 - m * p1 
52                      Pgen = Pg[i] 
53                      if Pgen < p2: 
54                          totalcost[i] = m * Pgen + b 
55                          break 
56                      totalcost[i] = m * Pgen + b 
57

59   
60      return totalcost
