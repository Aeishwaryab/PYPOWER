 30      """Evaluates objective function, gradient and Hessian for OPF. 
 31   
 32      Objective function evaluation routine for AC optimal power flow, 
 33      suitable for use with L{pips}. Computes objective function value, 
 34      gradient and Hessian. 
 35   
 36      @param x: optimization vector 
 37      @param om: OPF model object 
 38   
 39      @return: C{F} - value of objective function. C{df} - (optional) gradient 
 40      of objective function (column vector). C{d2f} - (optional) Hessian of 
 41      objective function (sparse matrix). 
 42   
 43      @see: L{opf_consfcn}, L{opf_hessfcn} 
 44      """ 
 45      ##----- initialize ----- 
 46      ## unpack data 
 47      ppc = om.get_ppc() 
 48      baseMVA, gen, gencost = ppc["baseMVA"], ppc["gen"], ppc["gencost"] 
 49      cp = om.get_cost_params() 
 50      N, Cw, H, dd, rh, kk, mm = \ 
 51          cp["N"], cp["Cw"], cp["H"], cp["dd"], cp["rh"], cp["kk"], cp["mm"] 
 52      vv, _, _, _ = om.get_idx() 
 53   
 54      ## problem dimensions 
 55      ng = gen.shape[0]          ## number of dispatchable injections 
 56      ny = om.getN('var', 'y')   ## number of piece-wise linear costs 
 57      nxyz = len(x)              ## total number of control vars of all types 
 58   
 59      ## grab Pg & Qg 
 60      Pg = x[vv["i1"]["Pg"]:vv["iN"]["Pg"]]  ## active generation in p.u. 
 61      Qg = x[vv["i1"]["Qg"]:vv["iN"]["Qg"]]  ## reactive generation in p.u. 
 62   
 63      ##----- evaluate objective function ----- 
 64      ## polynomial cost of P and Q 
 65      # use totcost only on polynomial cost in the minimization problem 
 66      # formulation, pwl cost is the sum of the y variables. 
 67      ipol = find(gencost[:, MODEL] == POLYNOMIAL)   ## poly MW and MVAr costs 
 68      xx = r_[ Pg, Qg ] * baseMVA 
 69      if any(ipol): 
 70          f = sum( totcost(gencost[ipol, :], xx[ipol]) )  ## cost of poly P or Q 
 71      else: 
 72          f = 0 
 73   
 74      ## piecewise linear cost of P and Q 
 75      if ny > 0: 
 76          ccost = sparse((ones(ny), 
 77                          (zeros(ny), arange(vv["i1"]["y"], vv["iN"]["y"]))), 
 78                         (1, nxyz)).toarray().flatten() 
 79          f = f + dot(ccost, x) 
 80      else: 
 81          ccost = zeros(nxyz) 
 82   
 83      ## generalized cost term 
 84      if issparse(N) and N.nnz > 0: 
 85          nw = N.shape[0] 
 86          r = N * x - rh                   ## Nx - rhat 
 87          iLT = find(r < -kk)              ## below dead zone 
 88          iEQ = find((r == 0) & (kk == 0)) ## dead zone doesn't exist 
 89          iGT = find(r > kk)               ## above dead zone 
 90          iND = r_[iLT, iEQ, iGT]          ## rows that are Not in the Dead region 
 91          iL = find(dd == 1)           ## rows using linear function 
 92          iQ = find(dd == 2)           ## rows using quadratic function 
 93          LL = sparse((ones(len(iL)), (iL, iL)), (nw, nw)) 
 94          QQ = sparse((ones(len(iQ)), (iQ, iQ)), (nw, nw)) 
 95          kbar = sparse((r_[ones(len(iLT)), zeros(len(iEQ)), -ones(len(iGT))], 
 96                         (iND, iND)), (nw, nw)) * kk 
 97          rr = r + kbar                  ## apply non-dead zone shift 
 98          M = sparse((mm[iND], (iND, iND)), (nw, nw))  ## dead zone or scale 
 99          diagrr = sparse((rr, (arange(nw), arange(nw))), (nw, nw)) 
100   
101          ## linear rows multiplied by rr(i), quadratic rows by rr(i)^2 
102          w = M * (LL + QQ * diagrr) * rr 
103   
104          f = f + dot(w * H, w) / 2 + dot(Cw, w) 
105   
106      ##----- evaluate cost gradient ----- 
107      ## index ranges 
108      iPg = range(vv["i1"]["Pg"], vv["iN"]["Pg"]) 
109      iQg = range(vv["i1"]["Qg"], vv["iN"]["Qg"]) 
110   
111      ## polynomial cost of P and Q 
112      df_dPgQg = zeros(2 * ng)        ## w.r.t p.u. Pg and Qg 
113      df_dPgQg[ipol] = baseMVA * polycost(gencost[ipol, :], xx[ipol], 1) 
114      df = zeros(nxyz) 
115      df[iPg] = df_dPgQg[:ng] 
116      df[iQg] = df_dPgQg[ng:ng + ng] 
117   
118      ## piecewise linear cost of P and Q 
119      df = df + ccost  # The linear cost row is additive wrt any nonlinear cost. 
120   
121      ## generalized cost term 
122      if issparse(N) and N.nnz > 0: 
123          HwC = H * w + Cw 
124          AA = N.T * M * (LL + 2 * QQ * diagrr) 
125          df = df + AA * HwC 
126   
127          ## numerical check 
128          if 0:    ## 1 to check, 0 to skip check 
129              ddff = zeros(df.shape) 
130              step = 1e-7 
131              tol  = 1e-3 
132              for k in range(len(x)): 
133                  xx = x 
134                  xx[k] = xx[k] + step 
135                  ddff[k] = (opf_costfcn(xx, om) - f) / step 
136              if max(abs(ddff - df)) > tol: 
137                  idx = find(abs(ddff - df) == max(abs(ddff - df))) 
138                  print 'Mismatch in gradient' 
139                  print 'idx             df(num)         df              diff' 
140                  print '%4d%16g%16g%16g' % \ 
141                      (range(len(df)), ddff.T, df.T, abs(ddff - df).T) 
142                  print 'MAX' 
143                  print '%4d%16g%16g%16g' % \ 
144                      (idx.T, ddff[idx].T, df[idx].T, abs(ddff[idx] - df[idx]).T) 
145   
146      if not return_hessian: 
147          return f, df 
148   
149      ## ---- evaluate cost Hessian ----- 
150      pcost = gencost[range(ng), :] 
151      if gencost.shape[0] > ng: 
152          qcost = gencost[ng + 1:2 * ng, :] 
153      else: 
154          qcost = array([]) 
155   
156      ## polynomial generator costs 
157      d2f_dPg2 = sparse((ng, 1))               ## w.r.t. p.u. Pg 
158      d2f_dQg2 = sparse((ng, 1))               ## w.r.t. p.u. Qg 
159      ipolp = find(pcost[:, MODEL] == POLYNOMIAL) 
160      d2f_dPg2[ipolp] = \ 
161              baseMVA**2 * polycost(pcost[ipolp, :], Pg[ipolp]*baseMVA, 2) 
162      if any(qcost):          ## Qg is not free 
163          ipolq = find(qcost[:, MODEL] == POLYNOMIAL) 
164          d2f_dQg2[ipolq] = \ 
165                  baseMVA**2 * polycost(qcost[ipolq, :], Qg[ipolq] * baseMVA, 2) 
166      i = r_[iPg, iQg].T 
167      d2f = sparse((r_[d2f_dPg2, d2f_dQg2], (i, i)), (nxyz, nxyz)) 
168   
169      ## generalized cost 
170      if any(N): 
171          d2f = d2f + AA * H * AA.T + 2 * N.T * M * QQ * \ 
172                  sparse((HwC, (range(nw), range(nw))), (nw, nw)) * N 
173   
174      return f, df, d2f
