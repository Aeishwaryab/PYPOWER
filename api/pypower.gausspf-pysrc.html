 27 -def gausspf(Ybus, Sbus, V0, ref, pv, pq, ppopt=None): 

 28      """Solves the power flow using a Gauss-Seidel method. 
 29   
 30      Solves for bus voltages given the full system admittance matrix (for 
 31      all buses), the complex bus power injection vector (for all buses), 
 32      the initial vector of complex bus voltages, and column vectors with 
 33      the lists of bus indices for the swing bus, PV buses, and PQ buses, 
 34      respectively. The bus voltage vector contains the set point for 
 35      generator (including ref bus) buses, and the reference angle of the 
 36      swing bus, as well as an initial guess for remaining magnitudes and 
 37      angles. C{ppopt} is a PYPOWER options vector which can be used to 
 38      set the termination tolerance, maximum number of iterations, and 
 39      output options (see C{ppoption} for details). Uses default options 
 40      if this parameter is not given. Returns the final complex voltages, 
 41      a flag which indicates whether it converged or not, and the number 
 42      of iterations performed. 
 43   
 44      @see: L{runpf} 
 45      """ 
 46      ## default arguments 
 47      if ppopt is None: 
 48          ppopt = ppoption() 
 49   
 50      ## options 
 51      tol     = ppopt['PF_TOL'] 
 52      max_it  = ppopt['PF_MAX_IT_GS'] 
 53      verbose = ppopt['VERBOSE'] 
 54   
 55      ## initialize 
 56      converged = 0 
 57      i = 0 
 58      V = V0.copy() 
 59      #Va = angle(V) 
 60      Vm = abs(V) 
 61   
 62      ## set up indexing for updating V 
 63      npv = len(pv) 
 64      npq = len(pq) 
 65      pvpq = r_[pv, pq] 
 66   
 67      ## evaluate F(x0) 
 68      mis = V * conj(Ybus * V) - Sbus 
 69      F = r_[  mis[pvpq].real, 
 70               mis[pq].imag   ] 
 71   
 72      ## check tolerance 
 73      normF = linalg.norm(F, Inf) 
 74      if verbose > 0: 
 75          sys.stdout.write('(Gauss-Seidel)\n') 
 76      if verbose > 1: 
 77          sys.stdout.write('\n it    max P & Q mismatch (p.u.)') 
 78          sys.stdout.write('\n----  ---------------------------') 
 79          sys.stdout.write('\n%3d        %10.3e' % (i, normF)) 
 80      if normF < tol: 
 81          converged = 1 
 82          if verbose > 1: 
 83              sys.stdout.write('\nConverged!\n') 
 84   
 85      ## do Gauss-Seidel iterations 
 86      while (not converged and i < max_it): 
 87          ## update iteration counter 
 88          i = i + 1 
 89   
 90          ## update voltage 
 91          ## at PQ buses 
 92          for k in pq[range(npq)]: 
 93              tmp = (conj(Sbus[k] / V[k]) - Ybus[k, :] * V) / Ybus[k, k] 
 94              V[k] = V[k] + asscalar(tmp) 
 95   
 96          ## at PV buses 
 97          if npv: 
 98              for k in pv[range(npv)]: 
 99                  tmp = (V[k] * conj(Ybus[k,:] * V)).imag 
100                  Sbus[k] = Sbus[k].real + 1j * asscalar(tmp) 
101                  tmp = (conj(Sbus[k] / V[k]) - Ybus[k, :] * V) / Ybus[k, k] 
102                  V[k] = V[k] + asscalar(tmp) 
103  #               V[k] = Vm[k] * V[k] / abs(V[k]) 
104              V[pv] = Vm[pv] * V[pv] / abs(V[pv]) 
105   
106          ## evalute F(x) 
107          mis = V * conj(Ybus * V) - Sbus 
108          F = r_[  mis[pv].real, 
109                   mis[pq].real, 
110                   mis[pq].imag  ] 
111   
112          ## check for convergence 
113          normF = linalg.norm(F, Inf) 
114          if verbose > 1: 
115              sys.stdout.write('\n%3d        %10.3e' % (i, normF)) 
116          if normF < tol: 
117              converged = 1 
118              if verbose: 
119                  sys.stdout.write('\nGauss-Seidel power flow converged in ' 
120                                   '%d iterations.\n' % i) 
121   
122      if verbose: 
123          if not converged: 
124              sys.stdout.write('Gauss-Seidel power did not converge in %d ' 
125                               'iterations.' % i) 
126   
127      return V, converged, i
