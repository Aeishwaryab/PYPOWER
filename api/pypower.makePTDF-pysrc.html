31      """Builds the DC PTDF matrix for a given choice of slack. 
32   
33      Returns the DC PTDF matrix for a given choice of slack. The matrix is 
34      C{nbr x nb}, where C{nbr} is the number of branches and C{nb} is the 
35      number of buses. The C{slack} can be a scalar (single slack bus) or an 
36      C{nb x 1} column vector of weights specifying the proportion of the 
37      slack taken up at each bus. If the C{slack} is not specified the 
38      reference bus is used by default. 
39   
40      For convenience, C{slack} can also be an C{nb x nb} matrix, where each 
41      column specifies how the slack should be handled for injections 
42      at that bus. 
43   
44      @see: L{makeLODF} 
45      """ 
46      ## use reference bus for slack by default 
47      if slack is None: 
48          slack = find(bus[:, BUS_TYPE] == REF) 
49          slack = slack[0] 
50   
51      ## set the slack bus to be used to compute initial PTDF 
52      if isscalar(slack): 
53          slack_bus = slack 
54      else: 
55          slack_bus = 0      ## use bus 1 for temp slack bus 
56   
57      nb = bus.shape[0] 
58      nbr = branch.shape[0] 
59      noref = arange(1, nb)      ## use bus 1 for voltage angle reference 
60      noslack = find(arange(nb) != slack_bus) 
61   
62      ## check that bus numbers are equal to indices to bus (one set of bus numbers) 
63      if any(bus[:, BUS_I] != arange(nb)): 
64          stderr.write('makePTDF: buses must be numbered consecutively') 
65   
66      ## compute PTDF for single slack_bus 
67      Bbus, Bf, _, _ = makeBdc(baseMVA, bus, branch) 
68      Bbus, Bf = Bbus.todense(), Bf.todense() 
69      H = zeros((nbr, nb)) 
70      H[:, noslack] = solve( Bbus[ix_(noslack, noref)].T, Bf[:, noref].T ).T 
71      #             = Bf[:, noref] * inv(Bbus[ix_(noslack, noref)]) 
72   
73      ## distribute slack, if requested 
74      if not isscalar(slack): 
75          if len(slack.shape) == 1:  ## slack is a vector of weights 
76              slack = slack / sum(slack)   ## normalize weights 
77   
78              ## conceptually, we want to do ... 
79              ##    H = H * (eye(nb, nb) - slack * ones((1, nb))) 
80              ## ... we just do it more efficiently 
81              v = dot(H, slack) 
82              for k in range(nb): 
83                  H[:, k] = H[:, k] - v 
84          else: 
85              H = dot(H, slack) 
86   
87      return H
