30      """Builds the B matrices and phase shift injections for DC power flow. 
31   
32      Returns the B matrices and phase shift injection vectors needed for a 
33      DC power flow. 
34      The bus real power injections are related to bus voltage angles by:: 
35          P = Bbus * Va + PBusinj 
36      The real power flows at the from end the lines are related to the bus 
37      voltage angles by:: 
38          Pf = Bf * Va + Pfinj 
39      Does appropriate conversions to p.u. 
40   
41      @see: L{dcpf} 
42      """ 
43      ## constants 
44      nb = bus.shape[0]          ## number of buses 
45      nl = branch.shape[0]       ## number of lines 
46   
47      ## check that bus numbers are equal to indices to bus (one set of bus nums) 
48      if any(bus[:, BUS_I] != range(nb)): 
49          stderr.write('makeBdc: buses must be numbered consecutively in ' 
50                       'bus matrix\n') 
51   
52      ## for each branch, compute the elements of the branch B matrix and the phase 
53      ## shift "quiescent" injections, where 
54      ## 
55      ##      | Pf |   | Bff  Bft |   | Vaf |   | Pfinj | 
56      ##      |    | = |          | * |     | + |       | 
57      ##      | Pt |   | Btf  Btt |   | Vat |   | Ptinj | 
58      ## 
59      stat = branch[:, BR_STATUS]               ## ones at in-service branches 
60      b = stat / branch[:, BR_X]                ## series susceptance 
61      tap = ones(nl)                            ## default tap ratio = 1 
62      i = find(branch[:, TAP])               ## indices of non-zero tap ratios 
63      tap[i] = branch[i, TAP]                   ## assign non-zero tap ratios 
64      b = b / tap 
65   
66      ## build connection matrix Cft = Cf - Ct for line and from - to buses 
67      f = branch[:, F_BUS]                           ## list of "from" buses 
68      t = branch[:, T_BUS]                           ## list of "to" buses 
69      i = r_[range(nl), range(nl)]                   ## double set of row indices 
70      ## connection matrix 
71      Cft = sparse((r_[ones(nl), -ones(nl)], (i, r_[f, t])), (nl, nb)) 
72   
73      ## build Bf such that Bf * Va is the vector of real branch powers injected 
74      ## at each branch's "from" bus 
75      Bf = sparse((r_[b, -b], (i, r_[f, t])))## = spdiags(b, 0, nl, nl) * Cft 
76   
77      ## build Bbus 
78      Bbus = Cft.T * Bf 
79   
80      ## build phase shift injection vectors 
81      Pfinj = b * (-branch[:, SHIFT] * pi / 180)  ## injected at the from bus ... 
82      # Ptinj = -Pfinj                            ## and extracted at the to bus 
83      Pbusinj = Cft.T * Pfinj                ## Pbusinj = Cf * Pfinj + Ct * Ptinj 
84   
85      return Bbus, Bf, Pbusinj, Pfinj
