 44      """Solves a DC optimal power flow. 
 45   
 46      Inputs are an OPF model object, a PYPOWER options dict and 
 47      a dict containing fields (can be empty) for each of the desired 
 48      optional output fields. 
 49   
 50      Outputs are a C{results} dict, C{success} flag and C{raw} output dict. 
 51   
 52      C{results} is a PYPOWER case dict (ppc) with the usual baseMVA, bus 
 53      branch, gen, gencost fields, along with the following additional 
 54      fields: 
 55          - C{order}      see 'help ext2int' for details of this field 
 56          - C{x}          final value of optimization variables (internal order) 
 57          - C{f}          final objective function value 
 58          - C{mu}         shadow prices on ... 
 59              - C{var} 
 60                  - C{l}  lower bounds on variables 
 61                  - C{u}  upper bounds on variables 
 62              - C{lin} 
 63                  - C{l}  lower bounds on linear constraints 
 64                  - C{u}  upper bounds on linear constraints 
 65          - C{g}          (optional) constraint values 
 66          - C{dg}         (optional) constraint 1st derivatives 
 67          - C{df}         (optional) obj fun 1st derivatives (not yet implemented) 
 68          - C{d2f}        (optional) obj fun 2nd derivatives (not yet implemented) 
 69   
 70      C{success} is C{True} if solver converged successfully, C{False} otherwise. 
 71   
 72      C{raw} is a raw output dict in form returned by MINOS 
 73          - C{xr}     final value of optimization variables 
 74          - C{pimul}  constraint multipliers 
 75          - C{info}   solver specific termination code 
 76          - C{output} solver specific output information 
 77   
 78      @see: L{opf}, L{qps_pypower} 
 79      """ 
 80      if out_opt is None: 
 81          out_opt = {} 
 82   
 83      ## options 
 84      verbose = ppopt['VERBOSE'] 
 85      alg     = ppopt['OPF_ALG_DC'] 
 86   
 87      if alg == 0: 
 88          alg = 200  # Use PIPS 
 89   
 90      ## unpack data 
 91      ppc = om.get_ppc() 
 92      baseMVA, bus, gen, branch, gencost = \ 
 93          ppc["baseMVA"], ppc["bus"], ppc["gen"], ppc["branch"], ppc["gencost"] 
 94      cp = om.get_cost_params() 
 95      N, H, Cw = cp["N"], cp["H"], cp["Cw"] 
 96      fparm = array(c_[cp["dd"], cp["rh"], cp["kk"], cp["mm"]]) 
 97      Bf = om.userdata('Bf') 
 98      Pfinj = om.userdata('Pfinj') 
 99      vv, ll, _, _ = om.get_idx() 
100   
101      ## problem dimensions 
102      ipol = find(gencost[:, MODEL] == POLYNOMIAL) ## polynomial costs 
103      ipwl = find(gencost[:, MODEL] == PW_LINEAR)  ## piece-wise linear costs 
104      nb = bus.shape[0]              ## number of buses 
105      nl = branch.shape[0]           ## number of branches 
106      nw = N.shape[0]                ## number of general cost vars, w 
107      ny = om.getN('var', 'y')       ## number of piece-wise linear costs 
108      nxyz = om.getN('var')          ## total number of control vars of all types 
109   
110      ## linear constraints & variable bounds 
111      A, l, u = om.linear_constraints() 
112      x0, xmin, xmax = om.getv() 
113   
114      ## set up objective function of the form: f = 1/2 * X'*HH*X + CC'*X 
115      ## where X = [x;y;z]. First set up as quadratic function of w, 
116      ## f = 1/2 * w'*HHw*w + CCw'*w, where w = diag(M) * (N*X - Rhat). We 
117      ## will be building on the (optionally present) user supplied parameters. 
118   
119      ## piece-wise linear costs 
120      any_pwl = int(ny > 0) 
121      if any_pwl: 
122          # Sum of y vars. 
123          Npwl = sparse((ones(ny), (zeros(ny), arange(vv["i1"]["y"], vv["iN"]["y"]))), (1, nxyz)) 
124          Hpwl = sparse((1, 1)) 
125          Cpwl = array([1]) 
126          fparm_pwl = array([[1, 0, 0, 1]]) 
127      else: 
128          Npwl = None#zeros((0, nxyz)) 
129          Hpwl = None#array([]) 
130          Cpwl = array([]) 
131          fparm_pwl = zeros((0, 4)) 
132   
133      ## quadratic costs 
134      npol = len(ipol) 
135      if any(find(gencost[ipol, NCOST] > 3)): 
136          stderr.write('DC opf cannot handle polynomial costs with higher ' 
137                       'than quadratic order.\n') 
138      iqdr = find(gencost[ipol, NCOST] == 3) 
139      ilin = find(gencost[ipol, NCOST] == 2) 
140      polycf = zeros((npol, 3))         ## quadratic coeffs for Pg 
141      if len(iqdr) > 0: 
142          polycf[iqdr, :] = gencost[ipol[iqdr], COST:COST + 3] 
143      if npol: 
144          polycf[ilin, 1:3] = gencost[ipol[ilin], COST:COST + 2] 
145      polycf = dot(polycf, diag([ baseMVA**2, baseMVA, 1]))     ## convert to p.u. 
146      if npol: 
147          Npol = sparse((ones(npol), (arange(npol), vv["i1"]["Pg"] + ipol)), 
148                        (npol, nxyz))  # Pg vars 
149          Hpol = sparse((2 * polycf[:, 0], (arange(npol), arange(npol))), 
150                        (npol, npol)) 
151      else: 
152          Npol = None 
153          Hpol = None 
154      Cpol = polycf[:, 1] 
155      fparm_pol = ones((npol, 1)) * array([[1, 0, 0, 1]]) 
156   
157      ## combine with user costs 
158      NN = vstack([n for n in [Npwl, Npol, N] if n is not None and n.shape[0] > 0], "csr") 
159      # FIXME: Zero dimension sparse matrices. 
160      if (Hpwl is not None) and any_pwl and (npol + nw): 
161          Hpwl = hstack([Hpwl, sparse((any_pwl, npol + nw))]) 
162      if Hpol is not None: 
163          if any_pwl and npol: 
164              Hpol = hstack([sparse((npol, any_pwl)), Hpol]) 
165          if npol and nw: 
166              Hpol = hstack([Hpol, sparse((npol, nw))]) 
167      if (H is not None) and nw and (any_pwl + npol): 
168          H = hstack([sparse((nw, any_pwl + npol)), H]) 
169      HHw = vstack([h for h in [Hpwl, Hpol, H] if h is not None and h.shape[0] > 0], "csr") 
170      CCw = r_[Cpwl, Cpol, Cw] 
171      ffparm = r_[fparm_pwl, fparm_pol, fparm] 
172   
173      ## transform quadratic coefficients for w into coefficients for X 
174      nnw = any_pwl + npol + nw 
175      M = sparse((ffparm[:, 3], (range(nnw), range(nnw)))) 
176      MR = M * ffparm[:, 1] 
177      HMR = HHw * MR 
178      MN = M * NN 
179      HH = MN.T * HHw * MN 
180      CC = MN.T * (CCw - HMR) 
181      C0 = 0.5 * dot(MR, HMR) + sum(polycf[:, 2])  # Constant term of cost. 
182   
183      ## set up input for QP solver 
184      opt = {'alg': alg, 'verbose': verbose} 
185      if (alg == 200) or (alg == 250): 
186          ## try to select an interior initial point 
187          Varefs = bus[bus[:, BUS_TYPE] == REF, VA] * (pi / 180.0) 
188   
189          lb, ub = xmin.copy(), xmax.copy() 
190          lb[xmin == -Inf] = -1e10   ## replace Inf with numerical proxies 
191          ub[xmax ==  Inf] =  1e10 
192          x0 = (lb + ub) / 2; 
193          # angles set to first reference angle 
194          x0[vv["i1"]["Va"]:vv["iN"]["Va"]] = Varefs[0] 
195          if ny > 0: 
196              ipwl = find(gencost[:, MODEL] == PW_LINEAR) 
197              # largest y-value in CCV data 
198              c = gencost.flatten('F')[sub2ind(gencost.shape, ipwl, 
199                                  NCOST + 2 * gencost[ipwl, NCOST])] 
200              x0[vv["i1"]["y"]:vv["iN"]["y"]] = max(c) + 0.1 * abs(max(c)) 
201   
202          ## set up options 
203          feastol = ppopt['PDIPM_FEASTOL'] 
204          gradtol = ppopt['PDIPM_GRADTOL'] 
205          comptol = ppopt['PDIPM_COMPTOL'] 
206          costtol = ppopt['PDIPM_COSTTOL'] 
207          max_it  = ppopt['PDIPM_MAX_IT'] 
208          max_red = ppopt['SCPDIPM_RED_IT'] 
209          if feastol == 0: 
210              feastol = ppopt['OPF_VIOLATION']    ## = OPF_VIOLATION by default 
211          opt["pips_opt"] = {  'feastol': feastol, 
212                               'gradtol': gradtol, 
213                               'comptol': comptol, 
214                               'costtol': costtol, 
215                               'max_it':  max_it, 
216                               'max_red': max_red, 
217                               'cost_mult': 1  } 
218      elif alg == 400: 
219          opt['ipopt_opt'] = ipopt_options([], ppopt) 
220      elif alg == 500: 
221          opt['cplex_opt'] = cplex_options([], ppopt) 
222      elif alg == 600: 
223          opt['mosek_opt'] = mosek_options([], ppopt) 
224      else: 
225          raise ValueError, "Unrecognised solver [%d]." % alg 
226   
227      ##-----  run opf  ----- 
228      x, f, info, output, lmbda = \ 
229              qps_pypower(HH, CC, A, l, u, xmin, xmax, x0, opt) 
230      success = (info == 1) 
231   
232      ##-----  calculate return values  ----- 
233      if not any(isnan(x)): 
234          ## update solution data 
235          Va = x[vv["i1"]["Va"]:vv["iN"]["Va"]] 
236          Pg = x[vv["i1"]["Pg"]:vv["iN"]["Pg"]] 
237          f = f + C0 
238   
239          ## update voltages & generator outputs 
240          bus[:, VA] = Va * 180 / pi 
241          gen[:, PG] = Pg * baseMVA 
242   
243          ## compute branch flows 
244          branch[:, [QF, QT]] = zeros((nl, 2)) 
245          branch[:, PF] = (Bf * Va + Pfinj) * baseMVA 
246          branch[:, PT] = -branch[:, PF] 
247   
248      ## package up results 
249      mu_l = lmbda["mu_l"] 
250      mu_u = lmbda["mu_u"] 
251      muLB = lmbda["lower"] 
252      muUB = lmbda["upper"] 
253   
254      ## update Lagrange multipliers 
255      il = find((branch[:, RATE_A] != 0) & (branch[:, RATE_A] < 1e10)) 
256      bus[:, [LAM_P, LAM_Q, MU_VMIN, MU_VMAX]] = zeros((nb, 4)) 
257      gen[:, [MU_PMIN, MU_PMAX, MU_QMIN, MU_QMAX]] = zeros((gen.shape[0], 4)) 
258      branch[:, [MU_SF, MU_ST]] = zeros((nl, 2)) 
259      bus[:, LAM_P]       = (mu_u[ll["i1"]["Pmis"]:ll["iN"]["Pmis"]] - 
260                             mu_l[ll["i1"]["Pmis"]:ll["iN"]["Pmis"]]) / baseMVA 
261      branch[il, MU_SF]   = mu_u[ll["i1"]["Pf"]:ll["iN"]["Pf"]] / baseMVA 
262      branch[il, MU_ST]   = mu_u[ll["i1"]["Pt"]:ll["iN"]["Pt"]] / baseMVA 
263      gen[:, MU_PMIN]     = muLB[vv["i1"]["Pg"]:vv["iN"]["Pg"]] / baseMVA 
264      gen[:, MU_PMAX]     = muUB[vv["i1"]["Pg"]:vv["iN"]["Pg"]] / baseMVA 
265   
266      pimul = r_[ 
267        mu_l - mu_u, 
268       -ones((ny > 0)), ## dummy entry corresponding to linear cost row in A 
269        muLB - muUB 
270      ] 
271   
272      mu = { 'var': {'l': muLB, 'u': muUB}, 
273             'lin': {'l': mu_l, 'u': mu_u} } 
274   
275      results = deepcopy(ppc) 
276      results["bus"], results["branch"], results["gen"], \ 
277          results["om"], results["x"], results["mu"], results["f"] = \ 
278              bus, branch, gen, om, x, mu, f 
279   
280      raw = {'xr': x, 'pimul': pimul, 'info': info, 'output': output} 
281   
282      return results, success, raw
