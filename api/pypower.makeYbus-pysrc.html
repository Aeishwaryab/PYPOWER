29 -def makeYbus(baseMVA, bus, branch): 

30      """Builds the bus admittance matrix and branch admittance matrices. 
31   
32      Returns the full bus admittance matrix (i.e. for all buses) and the 
33      matrices C{Yf} and C{Yt} which, when multiplied by a complex voltage 
34      vector, yield the vector currents injected into each line from the 
35      "from" and "to" buses respectively of each line. Does appropriate 
36      conversions to p.u. 
37   
38      @see: L{makeSbus} 
39      """ 
40      ## constants 
41      nb = bus.shape[0]          ## number of buses 
42      nl = branch.shape[0]       ## number of lines 
43   
44      ## check that bus numbers are equal to indices to bus (one set of bus nums) 
45      if any(bus[:, BUS_I] != range(nb)): 
46          stderr.write('buses must appear in order by bus number\n') 
47   
48      ## for each branch, compute the elements of the branch admittance matrix where 
49      ## 
50      ##      | If |   | Yff  Yft |   | Vf | 
51      ##      |    | = |          | * |    | 
52      ##      | It |   | Ytf  Ytt |   | Vt | 
53      ## 
54      stat = branch[:, BR_STATUS]              ## ones at in-service branches 
55      Ys = stat / (branch[:, BR_R] + 1j * branch[:, BR_X])  ## series admittance 
56      Bc = stat * branch[:, BR_B]              ## line charging susceptance 
57      tap = ones(nl)                           ## default tap ratio = 1 
58      i = nonzero(branch[:, TAP])              ## indices of non-zero tap ratios 
59      tap[i] = branch[i, TAP]                  ## assign non-zero tap ratios 
60      tap = tap * exp(1j * pi / 180 * branch[:, SHIFT]) ## add phase shifters 
61   
62      Ytt = Ys + 1j * Bc / 2 
63      Yff = Ytt / (tap * conj(tap)) 
64      Yft = - Ys / conj(tap) 
65      Ytf = - Ys / tap 
66   
67      ## compute shunt admittance 
68      ## if Psh is the real power consumed by the shunt at V = 1.0 p.u. 
69      ## and Qsh is the reactive power injected by the shunt at V = 1.0 p.u. 
70      ## then Psh - j Qsh = V * conj(Ysh * V) = conj(Ysh) = Gs - j Bs, 
71      ## i.e. Ysh = Psh + j Qsh, so ... 
72      ## vector of shunt admittances 
73      Ysh = (bus[:, GS] + 1j * bus[:, BS]) / baseMVA 
74   
75      ## build connection matrices 
76      f = branch[:, F_BUS]                           ## list of "from" buses 
77      t = branch[:, T_BUS]                           ## list of "to" buses 
78      ## connection matrix for line & from buses 
79      Cf = csr_matrix((ones(nl), (range(nl), f)), (nl, nb)) 
80      ## connection matrix for line & to buses 
81      Ct = csr_matrix((ones(nl), (range(nl), t)), (nl, nb)) 
82   
83      ## build Yf and Yt such that Yf * V is the vector of complex branch currents injected 
84      ## at each branch's "from" bus, and Yt is the same for the "to" bus end 
85      i = r_[range(nl), range(nl)]                   ## double set of row indices 
86   
87      Yf = csr_matrix((r_[Yff, Yft], (i, r_[f, t])), (nl, nb)) 
88      Yt = csr_matrix((r_[Ytf, Ytt], (i, r_[f, t])), (nl, nb)) 
89      # Yf = spdiags(Yff, 0, nl, nl) * Cf + spdiags(Yft, 0, nl, nl) * Ct 
90      # Yt = spdiags(Ytf, 0, nl, nl) * Cf + spdiags(Ytt, 0, nl, nl) * Ct 
91   
92      ## build Ybus 
93      Ybus = Cf.T * Yf + Ct.T * Yt + \ 
94          csr_matrix((Ysh, (range(nb), range(nb))), (nb, nb)) 
95   
96      return Ybus, Yf, Yt
