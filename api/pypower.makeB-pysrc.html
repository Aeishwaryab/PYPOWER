28 -def makeB(baseMVA, bus, branch, alg): 

29      """Builds the FDPF matrices, B prime and B double prime. 
30   
31      Returns the two matrices B prime and B double prime used in the fast 
32      decoupled power flow. Does appropriate conversions to p.u. C{alg} is the 
33      value of the C{PF_ALG} option specifying the power flow algorithm. 
34   
35      @see: L{fdpf} 
36      """ 
37      ## constants 
38      nb = bus.shape[0]          ## number of buses 
39      nl = branch.shape[0]       ## number of lines 
40   
41      ##-----  form Bp (B prime)  ----- 
42      temp_branch = copy(branch)                 ## modify a copy of branch 
43      temp_bus = copy(bus)                       ## modify a copy of bus 
44      temp_bus[:, BS] = zeros(nb)                ## zero out shunts at buses 
45      temp_branch[:, BR_B] = zeros(nl)           ## zero out line charging shunts 
46      temp_branch[:, TAP] = ones(nl)             ## cancel out taps 
47      if alg == 2:                               ## if XB method 
48          temp_branch[:, BR_R] = zeros(nl)       ## zero out line resistance 
49      Bp = -1 * makeYbus(baseMVA, temp_bus, temp_branch)[0].imag 
50   
51      ##-----  form Bpp (B double prime)  ----- 
52      temp_branch = copy(branch)                 ## modify a copy of branch 
53      temp_branch[:, SHIFT] = zeros(nl)          ## zero out phase shifters 
54      if alg == 3:                               ## if BX method 
55          temp_branch[:, BR_R] = zeros(nl)    ## zero out line resistance 
56      Bpp = -1 * makeYbus(baseMVA, bus, temp_branch)[0].imag 
57   
58      return Bp, Bpp
